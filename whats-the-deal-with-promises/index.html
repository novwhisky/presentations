                var xhr = new XMLHttpRequest();

                xhr.onload = function( data ) {
                    // handle response
                };

                xhr.onerror = function( error ) {
                    // handle error
                };

                xhr.open("GET", "http://example.com", true);
                xhr.send();

TL;DR

I can has $.ajax() ?

                
                    var xhr = $.ajax({
                        url: "http://example.com",
                        type: "GET",
                        success: function( data ) {
                            // handle response
                        },
                        error: function( error ) {
                            // handle error
                        }
                    });
